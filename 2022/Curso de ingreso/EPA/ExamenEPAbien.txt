programa Cap2Completo 
procesos
  proceso contarCalle (ES esquinaConPapel:boolean ES floresEsq:numero)
  comenzar
    mientras HayFlorEnLaEsquina
      tomarFlor
      floresEsq:= floresEsq + 1
    si HayPapelEnLaEsquina
      esquinaConPapel:= V
    sino
      esquinaConPapel:= F
  fin
areas
  ciudad: AreaC(1,1,100,100)
robots
  robot robot1
  variables
    buscarEsquina: boolean
    floresEsqT: numero
  comenzar 
    buscarEsquina:= V
    floresEsqT:= 0
    derecha
    mientras (buscarEsquina) & (PosCa<100)
      mientras (buscarEsquina) & (PosAv<100)
        contarCalle (buscarEsquina, floresEsqT)
        mover
      si buscarEsquina
        contarCalle (buscarEsquina, floresEsqT)
        Pos (1, PosCa + 1)
      Informar (floresEsqT)
      floresEsqT:= 0
    mientras (buscarEsquina) & (PosAv<100)
      contarCalle (buscarEsquina, floresEsqT)
      mover
    si buscarEsquina
      contarCalle (buscarEsquina, floresEsqT)
    Informar (floresEsqT)
  fin
variables
  R-info: robot1
comenzar
  AsignarArea(R-info,ciudad)  
  Iniciar(R-info,1,1)
fin
