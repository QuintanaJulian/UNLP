procedure leerNumero ( numero: integer);
	begin
		if (numero <> 0) then begin
			writeln (numero mod 10);		//Lee al contrario
			leerNumero (numero div 10); 
			writeln (numero mod 10 );	//Lee en orden
		end;
end;
