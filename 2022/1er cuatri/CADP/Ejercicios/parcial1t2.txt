procedure insertarOrdenado (VAR pI: lista; dato: clientes);
var	
	nuevo, actual, ant: lista;	
begin
	new (nuevo); nuevo^.dato:= dato; nuevo^.sig:= nil;
	actual:= pI; ant:= pI;
	
	while (actual <> nil) and (actual^.dato < nuevo^.dato) do begin
		ant:= actual;
		actual:= actual^.sig;
		
		end;
	if (pI = actual) then begin
		actual^.sig:= pI;
		pI:= actual;
		end
	else begin
		nuevo^.sig := ant^.sig;
		ant^.sig := nuevo;
	end;

end;
