Process Profesor [ID = 0..2]{
	int corrigiendo;
	<Await (presentes = N); empezar = true>
	
	while (presentes <> 0){
		<Await (not listaOcupada); listaOcupada = true>

			<Pop (entregaron, corrigiendo)>
			listaOcupada = false;
			//corigiendo a corrigiendo

		}
	}
