programa hola
procesos
  proceso contarFlor1 (ES contarFlor : numero)
  comenzar
    contarFlor:= 0
    mientras HayFlorEnLaEsquina & (contarFlor <6)
      tomarFlor
      contarFlor:= contarFlor + 1
    repetir (contarFlor)
      depositarFlor 
  fin
  proceso contarPapel1 (ES contarPapel: numero)
  comenzar
    contarPapel:= 0
    mientras HayPapelEnLaEsquina & (contarPapel < 3)
      tomarPapel
      contarPapel:= contarPapel + 1
    repetir (contarPapel)
      depositarPapel
  fin
  proceso esquina (S esquinaV: boolean ; ES  callesEsquinaF: numero)
  variables
    contarFlor3: numero
    contarPapel3: numero
  comenzar
    contarFlor1(contarFlor3)
    contarPapel1(contarPapel3)
    esquinaV:= (contarFlor3 >=5) & (contarPapel3 = 2)
    si esquinaV
      mientras HayFlorEnLaEsquina
        tomarFlor
      mientras HayPapelEnLaEsquina
        tomarPapel
    sino
      callesEsquinaF:= (callesEsquinaF + 1 ) 
  fin
areas
  ciudad: AreaC (1,1,100,100)
robots
  robot robot1
  variables
    esquina2: numero
    callesEsquinaF1: numero
    totalEsquinaF: numero
    contarFlor4: numero
    contarPapel4: numero
    base, altura, separacion: numero
  comenzar
    base:= 99
    altura:= 99
    separacion:= 2
    derecha
    contarFlor1 (contarFlor4)
    contarPapel1 (contarPapel4)
    esquina (esquina2 , callesEsquinaF1)
    repetir 9999
      si (PosAv <= base)
        mover
        contarFlor1 (contarFlor4)
        contarPapel1 (contarPapel4)
        esquina (esquina2, callesEsquinaF1)
      sino
        si (PosCa < altura)
          Informar (callesEsquinaF1)
          totalEsquinaF:= totalEsquinaF + callesEsquinaF1
          callesEsquinaF1:= 0
          Pos (1, PosCa + separacion)
          contarFlor1 (contarFlor4)
          contarPapel1 (contarPapel4)
          esquina (esquina2, callesEsquinaF1)
        sino
          si (PosAv<= base)
            mover
            contarFlor1 (contarFlor4)
            contarPapel1 (contarPapel4)
            esquina (esquina2, callesEsquinaF1)
    Informar (callesEsquinaF1)
    totalEsquinaF:= (totalEsquinaF + callesEsquinaF1)
    Informar (totalEsquinaF)
  fin
variables
  R-Info: robot1
comenzar
  AsignarArea (R-Info, ciudad)
  Iniciar (R-Info, 1,1)
fin
